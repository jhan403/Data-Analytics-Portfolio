What proportion of User Types are Subscriber or Customers?

# Wickham, H., & Grolemund, G. (2016). R for Data Science. O'Reilly Publishing.
# Creating a function to print the proportion of Subscribers to the total.
calc_proportion_subscribers = function(data, city_name) {
    user_type_table = table(data$User.Type)
    proportion_subscribers = user_type_table['Subscriber'] / sum(user_type_table)
    print(sprintf('The proportion of subscribers from %s is %.2f%%', city_name, proportion_subscribers * 100))
}

#Creating a function to create pie charts for the proportion of Subscribers and Customers.
create_pie_chart = function(df, city_name) {
    user_type_counts = df %>%
        filter(User.Type %in% c('Subscriber', 'Customer')) %>%
        group_by(User.Type) %>%
        summarise(count = n()) %>%
        mutate(percentage = count / sum(count) * 100)
    ggplot(user_type_counts, aes(x = "", y = percentage, fill = User.Type)) +
        geom_bar(stat = "identity", width = 1) +
        coord_polar(theta = "y") +
        labs(title = paste("Percentage of Subscribers and Customers in", city_name), x = "", y = "") +
        theme_void() +
        geom_text(aes(label = paste0(round(percentage, 1), "%")), 
              position = position_stack(vjust = 0.5))
}

# Creating a function to create a histogram of number of rentals by day of week.
create_histogram = function(df, city_name) {
    df$Start.Time = as.POSIXct(df$Start.Time, format='%Y-%m-%d %H:%M:%S')
    df = df %>% 
        mutate(day_of_week = weekdays(Start.Time))
    df$day_of_week = factor(df$day_of_week, levels = c('Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'))
    ggplot(df, aes(x = day_of_week)) +
        geom_histogram(stat = 'count', fill = 'blue', color = 'black') +
        labs(title = paste('Histogram of Rentals in', city_name, 'by Day of the Week'), x = 'Day of the Week', y = 'Number of Rentals')
}

# Creating a function to create a table of number of rentals by day of week.
create_table = function(df, city_name) {
    df$Start.Time <- as.POSIXct(df$Start.Time, format='%Y-%m-%d %H:%M:%S')
    
    df <- df %>%
        mutate(day_of_week = weekdays(Start.Time))
    
    table_df = df %>%
        group_by(day_of_week) %>%
        summarise(Count = n()) %>%
        arrange(match(day_of_week, c('Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday')))
    
    cat('Table of Rentals in', city_name, 'by Day of Week')
    print(table_df)
    
}
